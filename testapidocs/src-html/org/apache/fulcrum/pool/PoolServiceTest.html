025import static org.junit.jupiter.api.Assertions.assertTrue;
026
027import org.apache.fulcrum.testcontainer.BaseUnit5Test;
028import org.junit.jupiter.api.BeforeEach;
029import org.junit.jupiter.api.Test;
030
031
032/**
033 * Basic testing of the PoolService Component
034 *
035 * @author <a href="mailto:painter@apache.org">Jeffery Painter</a>
036 * @author Eric Pugh
037 * @author <a href="mailto:mcconnell@apache.org">Stephen McConnell</a>
038 *
039 */
040public class PoolServiceTest extends BaseUnit5Test 
041{
042        /** Default pool service **/
043        private PoolService poolService = null;
044
045        /**
046         * Perform pool service setup
047         * 
048         * @throws Exception generic exception
049         */
050        @BeforeEach
051        public void setUp() throws Exception 
052        {
053        setConfigurationFileName("src/test/TestComponentConfig.xml");
054        setRoleFileName("src/test/TestRoleConfig.xml");
055        poolService = (PoolService) this.lookup(PoolService.class.getName());
056        }
057
058        /**
059         * Class to test for Object getInstance(Class)
060         * 
061         * @throws PoolException generic exception
062         */
063        @Test
064        public void testGetInstanceClass() throws PoolException 
065        {
066                Object object = poolService.getInstance(StringBuilder.class);
067                assertTrue(object instanceof StringBuilder);
068
069        }
070
071        /**
072         * Test adding an instance to the pool
073         */
074        @Test
075        public void testPutInstance() 
076        {
077                String s = "I am a string";
078                assertEquals(0, poolService.getSize("java.lang.String"));
079                poolService.putInstance(s);
080                assertEquals(1, poolService.getSize("java.lang.String"));
081
082        }
083
084        /**
085         * Test altering pool capacity
086         */
087        @Test
088        public void testGetSetCapacity() 
089        {
090                assertEquals(128, poolService.getCapacity("java.lang.String"));
091                poolService.setCapacity("java.lang.String", 278);
092                assertEquals(278, poolService.getCapacity("java.lang.String"));
093
094        }
095
096        /**
097         * Test to determine current size of the pool
098         */
099        @Test
100        public void testGetSize() 
101        {
102                String s = "I am a string";
103                assertEquals(0, poolService.getSize("java.lang.String"));
104                poolService.putInstance(s);
105                assertEquals(1, poolService.getSize("java.lang.String"));
106
107        }
108
109        /**
110         * Class to test for void clearPool(String)
111         */
112        @Test
113        public void testClearPoolString() 
114        {
115                String s = "I am a string";
116                assertEquals(0, poolService.getSize("java.lang.String"));
117                
118                poolService.putInstance(s);
119                assertEquals(1, poolService.getSize("java.lang.String"));
120                
121                poolService.clearPool("java.lang.String");
122                assertEquals(0, poolService.getSize("java.lang.String"));
123        }
124
125        /**
126         * Class to test for void clearPool()
127         */
128        @Test
129        public void testClearPool() 
130        {
131                String s = "I am a string";
132                assertEquals(0, poolService.getSize("java.lang.String"));
133                
134                poolService.putInstance(s);
135                poolService.putInstance(new Double(32));
136                assertEquals(1, poolService.getSize("java.lang.String"));
137                
138                poolService.clearPool();
139                assertEquals(0, poolService.getSize("java.lang.String"));
140                assertEquals(0, poolService.getSize("java.lang.Double"));
141        }
142}
