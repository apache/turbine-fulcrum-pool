022/**
023 * The Pool Service extends the Factory Service by adding support for pooling
024 * instantiated objects. When a new instance is requested, the service first
025 * checks its pool if one is available. If the the pool is empty, a new object
026 * will be instantiated from the specified class. If only class name is given,
027 * the request to create an instance will be forwarded to the Factory Service.
028 *
029 * <p>
030 * For objects implementing the Recyclable interface, a recycle method will be
031 * called, when they are taken from the pool, and a dispose method, when they
032 * are returned to the pool.
033 * </p>
034 *
035 * @author <a href="mailto:ilkka.priha@simsoft.fi">Ilkka Priha</a>
036 * @author <a href="mailto:mcconnell@apache.org">Stephen McConnell</a>
037 * @version $Id$
038 */
039public interface PoolService 
040{
041        /** Avalon role - used to id the component within the manager */
042        String ROLE = PoolService.class.getName();
043
044        /**
045         * The default pool capacity.
046         */
047        public static final int DEFAULT_POOL_CAPACITY = 128;
048
049        /**
050         * Gets an instance of a specified class either from the pool or by
051         * instantiating from the class if the pool is empty.
052         *
053         * @param <T> type of the instance
054         * @param clazz the class.
055         * @return the instance.
056         * @throws PoolException if recycling fails.
057         */
058        public <T> T getInstance(Class<?> clazz) throws PoolException;
059
060        /**
061         * Gets an instance of a specified class either from the pool or by
062         * instantiating from the class if the pool is empty.
063         *
064         * @param <T>       type of the instance class
065         * @param clazz     the class.
066         * @param params    an array containing the parameters of the constructor.
067         * @param signature an array containing the signature of the constructor.
068         * @return the instance.
069         * @throws PoolException if recycling fails.
070         */
071        public <T> T getInstance(Class<?> clazz, Object params[], String signature[]) throws PoolException;
072
073        /**
074         * Puts a used object back to the pool. Objects implementing the Recyclable
075         * interface can provide a recycle method to be called when they are reused and
076         * a dispose method to be called when they are returned to the pool.
077         *
078         * @param instance the object instance to recycle.
079         * @return true if the instance was accepted.
080         */
081        public boolean putInstance(Object instance);
082
083        /**
084         * Gets the capacity of the pool for a named class.
085         *
086         * @param className the name of the class.
087         * @return total capacity
088         */
089        public int getCapacity(String className);
090
091        /**
092         * Sets the capacity of the pool for a named class. Note that the pool will be
093         * cleared after the change.
094         *
095         * @param className the name of the class.
096         * @param capacity  the new capacity.
097         */
098        public void setCapacity(String className, int capacity);
099
100        /**
101         * Gets the current size of the pool for a named class.
102         *
103         * @param className the name of the class
104         * @return the size of the pool for the class
105         */
106        public int getSize(String className);
107
108        /**
109         * Clears instances of a named class from the pool.
110         *
111         * @param className the name of the class.
112         */
113        public void clearPool(String className);
114
115        /**
116         * Clears all instances from the pool.
117         */
118        void clearPool();
119
120}
